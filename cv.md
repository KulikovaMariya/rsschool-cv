function absentVowel(x){
  const vowelArray = ['a', 'e', 'i', 'o', 'u'];
  const vowelSet = new Set();

  let result = -1;

  vowelArray.forEach(char => {
    if (x.includes(char)) {
      vowelSet.add(char);
    }
  });

  vowelArray.forEach((char, idx) => {
    if (!vowelSet.has(char)) {
      result = idx;
    }
  });

  return result;
}
